51 /*! \brief Handles substitution of a single specifier
52  *
53  * This takes a string containing a single format specifier
54  * and substitutes in the value. By this point, the type
55  * should already have been checked against the type specifier
56  * of the format.
57  *
58  * \throw std::runtime_error If there is a problem with the substitution
59  *
60  * \tparam T The type of data to substitute with
61  *
62  * \param [inout] fmt String with a single format specifier. Will
63  * be replaced with the formatted string
64  * \param [in] subst What to put in place of the specifier
65  */
66 template<typename T>
67 void handle_fmt_single_(std::string & fmt, T subst)
68 {
69  static const int bufsize = 256;
70 
71  // should be fine for most substitutions
72  char buf[bufsize];
73 
74  // Try to write to the buffer
75  const int n = snprintf(buf, bufsize, fmt.c_str(), subst);
76 
77  // If the return value is >= bufsize, then the buffer wasn't
78  // big enough. Then we use heap allocation
79  if(n >= bufsize)
80  {
81  const int neededsize = n+1; // includes null termination
82 
83  std::unique_ptr<char> hbuf(new char[static_cast<size_t>(neededsize)]);
84  char * hbufp = hbuf.get();
85 
86  const int n2 = snprintf(hbufp, static_cast<size_t>(neededsize),
87  fmt.c_str(), subst);
88 
89  // these two conditions signal success
90  if(n2 >= 0 && n2 < neededsize)
91  fmt = std::string(hbufp); // copies to the output
92 
93  if(n2 < 0 || n2 >= neededsize)
94  throw std::runtime_error(std::string("Error here: ") +
95  std::to_string(n2));
96  }
97  else
98  fmt = std::string(buf);
99 }
